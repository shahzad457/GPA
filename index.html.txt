    <script>
        function gradeToPoint(grade) {
            const grades = { 'A': 4.0, 'B+': 3.5, 'B': 3.0, 'C+': 2.5, 'C': 2.0, 'D': 1.0, 'F': 0.0 };
            return grades[grade.toUpperCase()] || -1;
        }

        function calculateGPA() {
            const subjects = ['OOP', 'ExpositoryWriting', 'DiscreteStructure', 'AppliedPhysics', 'Calculus', 'ICP', 'OOPLab'];
            const credits = [3, 3, 3, 3, 3, 2, 1];
            let totalPoints = 0, totalCredits = 18;
            
            for (let i = 0; i < subjects.length; i++) {
                let grade = document.getElementById(subjects[i]).value;
                let gradePoint = gradeToPoint(grade);
                
                if (gradePoint === -1) {
                    document.getElementById('result').innerText = 'Invalid grade entered. Use A, B+, B, C+, C, D, or F.';
                    return;
                }
                totalPoints += gradePoint * credits[i];
            }
            let gpa = totalPoints / totalCredits;
            document.getElementById('result').innerText = 'Your GPA is: ' + gpa.toFixed(2);
        }
    </script>
